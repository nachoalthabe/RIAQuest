/**
 * @class Component
 * @module Components
 */
var Component = new Class({
	Implements : [ Options, Events ],
	/**
	 * @protected
	 * @property app
	 * @type Application
	 * @default undefined
	 */
	app : undefined,
	initialize : function() {
	},
	/**
	 * @method setContext
	 * @param {ComponentContext} options
	 * @returns {Component}
	 */
	setContext : function(options) {
		if (options.id)
			this.id = options.id;
		if (options.folder)
			this.sourceFolder = options.folder;
		if (options.params)
			this.setOptions(options.params);
		if (options.app)
			this.app = options.app;
		if (options.name)
			this.Name = options.name;
		if (this.init)
			this.init();
		return this;
	},
	/**
	  * @method getStoreKey
	  * @param {String} key
	  * @returns {Object}
	  */
	getStoreKey : function(key) {
		return this._app.getStoreKey(key);
	},
	/**
	 * @method setStoreKey
	 * @param {String} key
	 * @param {Object} value
	 * @returns {Object}
	 */
	setStoreKey : function(key, value) {
